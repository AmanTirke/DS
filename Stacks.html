1. CREATE A STACK IMPLEMENTING THE OPERATIONS ON AN INTEGER ARRAY (PUSH, POP, PEEK, ISEMPTY, ISFULL) 

                
                    #include 

                        #define max 100
                        struct stack
                        {
                            int arr[max];
                            int top;
                        };
                        
                        void init(struct stack *stack1)
                        {
                            stack1->top = -1;
                        }
                        int isempty(const struct stack *stack1)
                        {
                            return stack1->top == -1;
                        }
                        
                        int isfull(const struct stack *stack1)
                        {
                            return stack1->top == max - 1;
                        }
                        
                        void push(struct stack *stack1, int value)
                        {
                            if (isfull(stack1))
                            {
                                printf("Stack overflow! Cannot push %d.\n", value);
                            }
                            else
                            {
                                stack1->arr[++stack1->top] = value;
                                printf("Pushed %d onto the stack.\n", value);
                            }
                        }
                        
                        int pop(struct stack *stack1)
                        {
                            if (isempty(stack1))
                            {
                                printf("Stack underflow! Cannot pop from an empty stack.\n");
                                return -1;
                            }
                            else
                            {
                                int popped = stack1->arr[stack1->top--];
                                printf("Popped %d from the stack.\n", popped);
                                return popped;
                            }
                        }
                        
                        int peek(const struct stack *stack1)
                        {
                            if (isempty(stack1))
                            {
                                printf("Cannot peek from an empty stack.\n");
                                return -1;
                            }
                            else
                            {
                                printf("Top element of the stack: %d\n", stack1->arr[stack1->top]);
                                return stack1->arr[stack1->top];
                            }
                        }
                        
                        int main()
                        {
                            struct stack mystack;
                            init(&mystack);
                        
                            push(&mystack, 10);
                            push(&mystack, 20);
                            push(&mystack, 30);
                            peek(&mystack);
                            pop(&mystack);
                            pop(&mystack);
                            peek(&mystack);
                            pop(&mystack);
                        
                            return 0;

                        struct node
                        {
                            int data;
                            struct node* next;
                        };
                        
                        struct Stack
                        {
                            struct node* top;
                        };
                        
                        void init(struct Stack* stack)
                        {
                            stack->top = NULL;
                        }
                        
                        int isempty(const struct Stack* stack)
                        {
                            return stack->top == NULL;
                        }
                        
                        void push(struct Stack* stack, int value)
                        {
                            struct node* nnode = (struct node*)malloc(sizeof(struct node));
                            if (nnode == NULL)
                            {
                                printf("Memory allocation failed. Cannot push %d.\n", value);
                                return;
                            }
                        
                            nnode->data = value;
                            nnode->next = stack->top;
                        
                            stack->top = nnode;
                        
                            printf("Pushed %d onto the stack.\n", value);
                        }
                        
                        int pop(struct Stack* stack)
                        {
                            if (isempty(stack))
                            {
                                printf("Stack underflow! Cannot pop from an empty stack.\n");
                                return 0;
                            }
                        
                            int popped = stack->top->data;
                            struct node* temp = stack->top;
                            stack->top = stack->top->next;
                            free(temp);
                        
                            printf("Popped %d from the stack.\n", popped);
                            return popped;
                        }
                        
                        int peek(const struct Stack* stack)
                        {
                            if (isempty(stack))
                            {
                                printf("Cannot peek from an empty stack.\n");
                                return 0;
                            }
                        
                            printf("Top integer of the stack: %d\n", stack->top->data);
                            return stack->top->data;
                        }

                        void init(struct Stack* stack, int size)
                        {
                            stack->top = NULL;
                            stack->size = size;
                            stack->current = 0;
                        }
                        
                        int isempty(const struct Stack* stack)
                        {
                            return stack->top == NULL;
                        }
                        
                        int isfull(const struct Stack* stack)
                        {
                            return stack->current == stack->size;

                        void push(struct Stack* stack, char value)

                            if (isfull(stack))
                            {
                                printf("Stack overflow! Cannot push %c.\n", value);
                                return;

                            if (nnode == NULL)

                                printf("Memory allocation failed. Cannot push %c.\n", value);
                                return;
                            }
                        
                            nnode->data = value;
                            nnode->next = stack->top;
                            stack->top = nnode;
                            stack->current++;
                        
                            printf("Pushed %c onto the stack.\n", value);
                        }
                        
                        char pop(struct Stack* stack)
                        {
                            if (isempty(stack))
                            {
                                printf("Stack underflow! Cannot pop from an empty stack.\n");
                                return '\0';
                            }
                        
                            struct node* temp = stack->top;
                            char popped = temp->data;
                        
                            stack->top = temp->next;
                            free(temp);
                        
                            stack->current--;
                        
                            printf("Popped %c from the stack.\n", popped);
                        
                            return popped;
                        }
                        
                        char peek(const struct Stack* stack)
                        {
                            if (isempty(stack))
                            {
                                printf("Cannot peek from an empty stack.\n");
                                return '\0';
                            }
                        
                            return stack->top->data;
                        }

                            pop(&stack);
                            pop(&stack);

                        int match(char opening, char closing) {
                            return (opening == '(' && closing == ')') ||
                                   (opening == '{' && closing == '}') ||
                                   (opening == '[' && closing == ']');
                        }
                        
                        int same(const char* brac)
                        {
                            struct stack stack;
                            init(&stack);
                        
                            for (int i = 0; brac[i] != '\0'; ++i)
                            {
                                if (brac[i] == '(' || brac[i] == '{' || brac[i] == '[')
                                {
                                    push(&stack, brac[i]);
                                } else if (brac[i] == ')' || brac[i] == '}' || brac[i] == ']')
                                {
                                    if (isempty(&stack) || !match(pop(&stack), brac[i]))
                                    {
                                        return 0;
                                    }
                                }
                            }
                        
                            return isempty(&stack);
                        }

                            if (isfull(ds))
                            {
                                printf("Stack overflow! Cannot push %d to stack 1.\n", val);

                            {
                                ds->arr[++ds->top1] = val;
                                printf("Pushed %d onto stack 1.\n", val);
                            }

                        {
                            if (isfull(ds))
                            {
                                printf("Stack overflow! Cannot push %d to stack 2.\n", val);

                            else
                            {
                                ds->arr[--ds->top2] = val;
                                printf("Pushed %d onto stack 2.\n", val);
                            }
                        }
                        
                        int pop1(struct dual* ds)
                        {
                            if (isempty1(ds))
                            {

                            {

                            else
                            {

                                printf("Popped %d from stack 2.\n", popped);

                        void sort(struct Stack* stack)
                        {
                            struct Stack temp;
                            init(&temp);
                        
                            while (!isempty(stack))
                            {
                                int tempVal = pop(stack);
                        
                                while (!isempty(&temp) && peek(&temp) > tempVal)
                                {
                                    push(stack, pop(&temp));
                                }
                        
                                push(&temp, tempVal);
                            }
                        
                            while (!isempty(&temp))
                            {
                                push(stack, pop(&temp));
                            }
                        }

                            push(&stack, -5);
                            push(&stack, 18);
                            push(&stack, 14);
                            push(&stack, -3);
                        
                            printf("Original Stack: ");
                            while (!isempty(&stack))
                            {
                                printf("%d ", pop(&stack));

                            while (!isempty(&stack))
                            {
                                printf("%d ", pop(&stack));
                            }

int opera(char ch)
{
    return (ch == '+'  ch == '-'  ch == ''  ch == '');
}

int pren(char ch)
{
    if (ch == '+'  ch == '-')
    {
        return 1;
    }
    else if (ch == ''  ch == '')
    {
        return 2;
    }
    return 0;

void infix(const char infix, char prefix)
{
    struct Stack stack;
    init(&stack);

    int inl = strlen(infix);
    int prei = 0;

    for (int i = inl - 1; i = 0; --i)
    {
        char ch = infix[i];

        if (isalnum((unsigned char)ch))
        {
            prefix[prei++] = ch;
        }
        else if (opera(ch))
        {
            while (!isempty(&stack) && pren(peek(&stack))  pren(ch))
            {
                prefix[prei++] = pop(&stack);
            }
            push(&stack, ch);
        }
        else if (ch == ')')
        {
            push(&stack, ch);
        }
        else if (ch == '(')
        {
            while (!isempty(&stack) && peek(&stack) != ')')
            {
                prefix[prei++] = pop(&stack);
            }
            pop(&stack);
        }
    }

    while (!isempty(&stack))
    {
        prefix[prei++] = pop(&stack);
    }

    prefix[prei] = '0';

    int start = 0;
    int end = prei - 1;
    while (start  end)
    {
        char temp = prefix[start];
        prefix[start] = prefix[end];
        prefix[end] = temp;
        start++;
        end--;
    }
}

13. IMPLEMENT THE TOWER OF HANOI USING STACKS.

                
                    #include 
                        #include 
                        
                        struct stack
                        {
                            int o;
                            unsigned p;
                            int* ar;
                        };
                        
                        struct stack* cstack(unsigned p)
                        {
                            struct stack* stack = (struct stack*)malloc(sizeof(struct stack));
                            stack->p = p;
                            stack->o = -1;
                            stack->ar = (int*)malloc(stack->p * sizeof(int));
                            return stack;
                        }
                        
                        int isempty(struct stack* stack)
                        {
                            return stack->o == -1;
                        }
                        
                        int isfull(struct stack* stack)
                        {
                            return stack->o == stack->p - 1;
                        }
                        
                        void push(struct stack* stack, int item)
                        {
                            if (isfull(stack))
                            {
                                printf("Stack Overflow\n");
                                exit(EXIT_FAILURE);
                            }
                            stack->ar[++stack->o] = item;
                        }
                        
                        int pop(struct stack* stack)
                        {
                            if (isempty(stack))
                            {
                                printf("Stack Underflow\n");
                                exit(EXIT_FAILURE);
                            }
                            return stack->ar[stack->o--];
                        }
                        
                        void move(char a, char b, int c)
                        {
                            printf("Move disk %d from %c to %c\n", c, a, b);
                        }
                        
                        void tower(int p, struct stack* s, struct stack* r, struct stack* d)
                        {
                            if (p == 1) {
                                int c = pop(s);
                                push(d, c);
                                move('A', 'C', c);
                                return;
                            }
                            tower(p - 1, s, d, r);
                            int c = pop(s);
                            push(d, c);
                            move('A', 'C', c);
                            tower(p - 1, r, s, d);
                        }
                        
                        int main()
                        {
                            int p;
                        
                            printf("Enter the number of disks: ");
                            scanf("%d", &p);
                        
                            struct stack* s = cstack(p);
                            struct stack* r = cstack(p);
                            struct stack* d = cstack(p);
                        
                            for (int i = p; i > 0; i--)
                            {
                                push(s, i);
                            }
                            tower(p, s, r, d);
                            return 0;
                        }
