                        struct node* insp(struct node* head, int value, int pos)
                        {
                            struct node* nnode = (struct node*)malloc(sizeof(struct node));
                            nnode->data = value;
                        
                            if (pos == 1)
                            {
                                nnode->next = head;
                                return nnode;

                            struct node* temp = head;

                            {

                            temp->next = nnode;
                        
                            return head;

                            return head;

                        struct node* dele(struct node* head)
                        {
                            if (head == NULL)
                            {
                                printf("List is empty. Cannot delete.\n");
                                return NULL;
                            }
                        
                            if (head->next == NULL)
                            {
                                free(head);
                                return NULL;
                            }

                            struct node* prev = NULL;

                                prev = temp;
                                temp = temp->next;
                            }
